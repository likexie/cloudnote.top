# 类型判定

- API

  1. isWidgetType()判断是否是控件类型

     ```python
     from PyQt5.Qt import *
     class Window(QWidget):
         '''
         初始化窗口
         '''
     
         def __init__(self, title="python"):
             super().__init__()
             self.setWindowTitle(title)
             self.resize(500, 500)
             self.count = 1
     
         def QObject025(self):
             obj = QObject()
             w = QWidget()
             btn = QPushButton()
             label = QLabel()
             a = [obj,w,btn, label]
             for i in a:
                 print(i.isWidgetType())
     
         def setup_ui(self):
             self.QObject025()
             
     
     if __name__ == "__main__":
         import sys
         app = QApplication(sys.argv)
         window = Window(title='初始化')
         window.setup_ui()
         window.show()
         sys.exit(app.exec())
     # 输出结果
     False
     True
     True
     True
     ```

## 对象的删除

1. 删除方法obj.deleteLater()

   ```python
   from PyQt5.Qt import *
   
   
   class Window(QWidget):
       '''
       初始化窗口
       '''
   
       def __init__(self, title="python"):
           super().__init__()
           self.setWindowTitle(title)
           self.resize(500, 500)
           self.count = 1
   
       def QObject025(self):
           obj = QObject()
           w = QWidget()
           btn = QPushButton()
           label = QLabel()
           a = [obj,w,btn, label]
           for i in a:
               # print(i.isWidgetType())
               print(i.inherits("QPushButton"))
   
       def QObject026(self):
           '''
           类型判定
           '''
           label1 = QLabel(self)
           label1.setText("中云微迅")
           label1.move(100,100)
   
           label2 = QLabel(self)
           label2.setText("小米")
           label2.move(200,200)
   
           label3 = QLabel(self)
           label3.setText("华为")
           label3.move(300,300)
           # for i in self.findChildren(QLabel):
           #     i.setStyleSheet("color:red")
           
           for i in self.children():
               if i.inherits('QLabel'):
                   i.setStyleSheet("color:red")
   
       def QObject027(self):
           obj1 = QObject(self)
           obj2 = QObject()
           obj3 = QObject()
   
           obj2.setParent(obj1)
           obj3.setParent(obj2)  
   
           obj1.destroyed.connect(lambda : print("obj1被释放"))
           obj2.destroyed.connect(lambda : print("obj2被释放"))
           obj3.destroyed.connect(lambda : print("obj3被释放"))
           obj2.deleteLater()
   
       def setup_ui(self):
           self.QObject027()
           
   if __name__ == "__main__":
       import sys
       app = QApplication(sys.argv)
       window = Window(title='初始化')
       window.setup_ui()
       window.show()
       sys.exit(app.exec())
       
   # 输出结果
   obj2被释放
   obj3被释放
   obj1被释放
   ```
