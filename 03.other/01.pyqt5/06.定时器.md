# 定时器

```python
from PyQt5.Qt import *
import sys


class MyLabel(QLabel):

    def __init__(self,*agrs, **kwargs):
        super().__init__(*agrs, **kwargs)
        self.count = str(10)
        self.move(100,100)
        self.setStyleSheet("font-size:30px;color:red")
        self.setText(str(self.count))
        # 获取定时器的id，用于停止定时器
        self.timer_id = self.startTimer(1000)

    def timerEvent(self, *agrs, **kwargs):
        # 通过text()获取当前的值
        current_sec = int(self.text())
        current_sec -= 1
        self.setText(str(current_sec))
        if current_sec == 0:
            self.killTimer(self.timer_id)


app = QApplication(sys.argv)
window = QWidget()
time_label = MyLabel(window)


window.resize(500,500)
window.show()
sys.exit(app.exec())
```



## 定时器窗口每秒钟窗口变化

```python
from PyQt5.Qt import *
import sys

class MyQWidget(QWidget):

    def __init__(self,**kwargs):
        super().__init__(**kwargs)
        self.resize(10,10)
        self.setWindowTitle("窗口变换")
        self.show()

    def timerEvent(self,evt):
        # 获取窗口的宽高
        width = self.width()
        height = self.height()
        if width <= 800:
            self.resize(width+2,height+2)


app = QApplication(sys.argv)
window = MyQWidget()
window.startTimer(100)



sys.exit(app.exec())
```
