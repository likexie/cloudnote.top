```shell
 [root@zhu]#mysql_secure_installation
```


## 2. 创建用户
1. 创建本地用户
```sql
// 创建用户
CREATE user '[用户名]'@'localhost'(本地访问Only) / '[用户名]'@'%'(可以外部访问) IDENTIFIED BY '[密码]'; 

// 授权

GRANT SELECT, INSERT, UPDATE, DELETE, CREATE ON [数据库名称].*（.* 表明授权作用域为该数据库下所有表）TO [用户名];

//指定数据库
grant all privileges on 想授权的数据库.* to '用户名'@'%';
//全部数据库
grant all privileges on *.* to '用户名'@'%';


// 立即启用修改
FLUSH PRIVILEGES;
// 取消用户的所有数据库（表）的所有权限授权
REVOKE ALL ON *.* FROM [用户名];
// 删除用户
DELETE FROM mysql.user WHERE user='[用户名]';
// 给用户修改密码
set password for 'django'@'%'=password("django#222");
```
